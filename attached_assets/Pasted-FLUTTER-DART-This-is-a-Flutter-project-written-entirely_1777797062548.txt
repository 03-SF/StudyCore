Paste into Firebase Console → Firestore → Rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    match /decks/{deckId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
      match /cards/{cardId} {
        allow read, write: if request.auth != null &&
          request.auth.uid == get(/databases/$(database)/documents/decks/$(deckId)).data.userId;
      }
    }
    match /studySessions/{sessionId} {
      allow read, write: if request.auth != null && request.auth.uid == resource.data.userId;
      allow create: if request.auth != null && request.auth.uid == request.resource.data.userId;
    }
    match /groups/{groupId} {
      allow read: if request.auth != null && request.auth.uid in resource.data.memberIds;
      allow create: if request.auth != null;
      allow update: if request.auth != null && request.auth.uid == resource.data.adminId;
      match /messages/{messageId} {
        allow read, create: if request.auth != null &&
          request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.memberIds;
        allow delete: if request.auth != null && request.auth.uid == resource.data.userId;
      }
      match /sharedDecks/{deckId} {
        allow read, create: if request.auth != null &&
          request.auth.uid in get(/databases/$(database)/documents/groups/$(groupId)).data.memberIds;
      }
    }
  }
}

7. Spaced Repetition Algorithm (SM-2)
FLUTTER DART	Implement as a plain Dart class in lib/services/srs_service.dart — no Flutter imports needed in this file.

// lib/services/srs_service.dart — pure Dart, no Flutter dependency
import '../models/card_model.dart';

class SrsService {
  CardModel processRating(CardModel card, int rating) {
    int repetitions = card.repetitions;
    double easeFactor = card.easeFactor;
    int interval = card.interval;

    if (rating >= 3) {
      if (repetitions == 0)      interval = 1;
      else if (repetitions == 1) interval = 6;
      else                       interval = (interval * easeFactor).round();
      repetitions += 1;
    } else {
      repetitions = 0; interval = 1;
    }

    easeFactor += 0.1 - (5 - rating) * (0.08 + (5 - rating) * 0.02);
    if (easeFactor < 1.3) easeFactor = 1.3;

    return card.copyWith(
      repetitions: repetitions, easeFactor: easeFactor, interval: interval,
      dueDate: DateTime.now().add(Duration(days: interval)),
      lastRating: rating,
    );
  }

  bool isDue(CardModel card) => DateTime.now().isAfter(card.dueDate);

  double calculateMastery(List<CardModel> cards) {
    if (cards.isEmpty) return 0.0;
    return cards.where((c) => c.interval >= 7).length / cards.length;
  }
}

12. State Management
FLUTTER DART	Use the provider Dart package. All providers are ChangeNotifier subclasses. Register in main.dart inside a MultiProvider widget wrapping the app.

•	AuthProvider (ChangeNotifier) — UserModel? currentUser, bool isLoading, String? errorMessage. Listen to FirebaseAuth.instance.authStateChanges().
•	DeckProvider (ChangeNotifier) — List<DeckModel> decks, int totalDueCards. Firestore stream on decks where userId==uid.
•	SessionProvider (ChangeNotifier) — study session state: List<CardModel> cards, int currentIndex, Map<String,int> ratings, DateTime? sessionStart.
•	GroupProvider (ChangeNotifier) — List<GroupModel> myGroups. Firestore stream.
•	ThemeNotifier (ChangeNotifier) — ThemeMode themeMode. Load/save via SharedPreferences.

13. Error Handling Rules
NEVER show raw Dart exceptions or stack traces to users. Wrap every async Dart operation in try/catch. Map Firebase error codes to readable English strings.

•	FirebaseAuthException codes: user-not-found → 'No account with that email'. wrong-password → 'Incorrect password'. email-already-in-use → 'Email already registered'. network-request-failed → 'Connection failed. Check your internet.'
•	FirebaseException on Firestore writes → 'Could not save. Check your connection.'
•	AiServiceException → 'AI generation failed. Please try again.'
•	image_picker PlatformException → 'Could not access camera/gallery. Check permissions.'
•	StreamBuilder error state: show EmptyState widget with retry callback.

14. Push Notifications
•	FCM (Firebase Cloud Messaging Flutter plugin) — group chat messages only.
•	On group join: FirebaseMessaging.instance.subscribeToTopic('group_$groupId').
•	On notification tap: use go_router to navigate to /groups/$groupId/chat.
•	Local reminders: flutter_local_notifications Dart plugin. Daily scheduled notification. Cancel + reschedule when Settings time changes.
•	Request permission: permission_handler Dart plugin on first post-signup launch.

•	Enable Firestore offline persistence in main.dart:
FirebaseFirestore.instance.settings = const Settings(
